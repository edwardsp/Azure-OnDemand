[cluster Slurm]
FormLayout = selectionpanel
Category = Schedulers

Autoscale = $Autoscale

    [[node defaults]]
    UsePublicNetwork = false
    Credentials = $Credentials    
    SubnetId = $SubnetId
    Region = $Region
    KeyPairLocation = ~/.ssh/cyclecloud.pem
    
        [[[configuration]]]
        slurm.version = $configuration_slurm_version
        # Disable normal NFS exports and mounts
        cyclecloud.mounts.shared.disabled = true
        cyclecloud.exports.shared.disabled = true
        cyclecloud.exports.shared.samba.enabled = false
        cyclecloud.mounts.sched.disabled = true
        cyclecloud.exports.sched.disabled = true
        cyclecloud.exports.sched.samba.enabled = false
        cyclecloud.exports.defaults.samba.enabled = false
        cshared.server.legacy_links_disabled = true
        ganglia.install	= false
        cuser.base_home_dir = /share/home

        [[[configuration cyclecloud.mounts.nfs_share]]]
        type = nfs
        mountpoint = /share
        export_path = $NFSExport
        address = $NFSAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp

        [[[configuration cyclecloud.mounts.nfs_apps]]]
        type = nfs
        mountpoint = /apps
        export_path = ${NFSExport}/apps
        address = $NFSAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp
 
        [[[configuration cyclecloud.mounts.nfs_data]]]
        type = nfs
        mountpoint = /data
        export_path = ${NFSExport}/data
        address = $NFSAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp
 

        [[[configuration cyclecloud.mounts.nfs_home]]]
        type = nfs
        mountpoint = /share/home
        export_path = ${NFSExport}/home
        address = $NFSAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp
 
        # Slurm.conf is at /apps/slurm.conf
        [[[configuration cyclecloud.mounts.nfs_sched]]]
        type = nfs
        mountpoint = /sched
        export_path = ${NFSExport}/apps
        address = $NFSAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp
 
        [[[configuration cyclecloud.mounts.nfs_scratch]]]
        type = nfs
        mountpoint = /scratch
        export_path = /mnt/resource/scratch
        address = $NFSAddress
        options = defaults,rw,hard,rsize=65536,wsize=65536,vers=3,tcp
 

        [[[cluster-init cyclecloud/slurm:default]]]
        run_list = recipe[slurm::execute]

    [[nodearray hpc]]
    MachineType = $HPCMachineType
    ImageName = $HPCImageName
    MaxCount = $MaxHPCExecuteCount
    Azure.MaxScalesetSize = $HPCMaxScalesetSize
    AdditionalClusterInitSpecs = $HPCClusterInitSpecs


        [[[configuration]]]
        slurm.autoscale = true
        slurm.default_partition = true
        slurm.hpc = true

        [[[cluster-init slurm:execute]]]

        [[[network-interface eth0]]]
        AssociatePublicIpAddress = false

        Config.Plugin = pico.widget.HtmlTemplateWidget

    [[parameter NFSExport]]
    Label = Shared Export
    Config.Label = Shared export to mount
    DefaultValue = /share

    DefaultValue = localhost

    Order = 10

    [[parameters Software]]
    Description = "Specify the scheduling software, and base OS installed on all nodes, and optionally the cluster-init and chef versions from your Locker."
    Order = 10
